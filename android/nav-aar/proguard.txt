# The MessageSchema buffer is only written to in Java, but is accessed using
# the native memory address via sun.misc.Unsafe or libcore.io.Memory.
# Without this, the "field/removal/writeonly" optimization would remove the
# field and cause the buffer to be deallocated prematurely.
-keepclassmembers class com.google.protobuf.MessageSchema {
  java.nio.ByteBuffer buffer;
}
